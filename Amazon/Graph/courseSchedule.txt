// Using DFS

func canFinish(numCourses int, prerequisites [][]int) bool {
    graph := make(map[int][]int)

    // build graph
    for _, p := range prerequisites {
        a, b := p[0], p[1]
        graph[b] = append(graph[b], a)
    }

    // 0 = unvisited, 1 = visiting, 2 = visited
    state := make([]int, numCourses)

    var dfs func(int) bool
    dfs = func(course int) bool {
        // if visiting → cycle
        if state[course] == 1 {
            return false
        }

        // if already visited → no need to check again
        if state[course] == 2 {
            return true
        }

        // mark as visiting
        state[course] = 1

        // explore neighbors
        for _, nei := range graph[course] {
            if !dfs(nei) {
                return false
            }
        }

        // mark as visited
        state[course] = 2
        return true
    }

    // check all nodes
    for i := 0; i < numCourses; i++ {
        if !dfs(i) {
            return false
        }
    }

    return true
}


// USING BFS (Topological sorting) Kahn's Algorithm
func canFinish(numCourses int, prerequisites [][]int) bool {
    graph := make(map[int][]int)
    indegree := make([]int, numCourses)

    // build graph
    for _, p := range prerequisites {
        a, b := p[0], p[1]
        graph[b] = append(graph[b], a)
        indegree[a]++
    }

    // queue for nodes with indegree 0
    queue := []int{}
    for i := 0; i < numCourses; i++ {
        if indegree[i] == 0 {
            queue = append(queue, i)
        }
    }

    visited := 0

    for len(queue) > 0 {
        course := queue[0]
        queue = queue[1:]
        visited++

        for _, nei := range graph[course] {
            indegree[nei]--
            if indegree[nei] == 0 {
                queue = append(queue, nei)
            }
        }
    }

    return visited == numCourses
}
